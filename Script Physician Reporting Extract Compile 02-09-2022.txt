Drop table  #tbl_001


Select
t2.*

into #tbl_001
from
(
Select Distinct
	t1.VisitID,
	t2.ProviderID,
	t2.SeqID, ---Added this data element om 10-25-2021
	t2.DateTime,
	t1.DischargeDateTime,
	t2.Code,
	t3.GrouperVersion

from AbstractData t1
left outer Join BarSurgicalProcedures t2
on t1.VisitID = t2.VisitID
left outer join BarVisitFinancialData4 t3 on
t1.VisitID = t3.VisitID
--where (t1.ProviderID = 'GEOGR'
--and t1.DateTime like '%2018%'
where t1.DischargeDateTime like '%2021%'
)t2



-- Notes: After further analysis TSG Millenium logic operates off the discharge date being the primary. In the event a discharge is not present the logic seem to use the surgical proc date. 02-2020
-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
--Creates the procedure code dataset

Select distinct
	t1.VisitID,
	t1.ProviderID,
	t1.DateTime,
	t1.SeqID, ---Add this 10-25-2021
	t1.Code,
	t2.Name
into #tbl_01
from #tbl_001 t1
left outer join DAbsProcedures t2 on
t1.Code = t2.ProcedureID and
t1.GrouperVersion = t2.GrouperVersionID

--Functional Dependency 

select
	t1.VisitID

from #tbl_01 t1
group by
	t1.VisitID
having count(*)>1

	t1.DiagnosisCodeID,

	t1.Name

inner join

(Select 
max(t1.RowUpdateDateTime) as MaxRowDate,

t1.DiagnosisCodeID
)t2 on
t1.DiagnosisCodeID = t2.DiagnosisCodeID and
t1.RowUpdateDateTime = t2.MaxRowDate
